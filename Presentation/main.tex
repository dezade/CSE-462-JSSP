\section{Problem Description}

\begin{frame}{Problem Definition}
    % The \textbf{Job Shop Scheduling Problem (JSSP)} is a combinatorial optimization problem aiming to allocate shared resources over time to competing tasks.

    % \vspace{0.5em}
    \begin{block}{Mathematical Formulation}
        \begin{itemize}[<+->]
            \item \textbf{Given}: A set of $N_J$ jobs $\mathcal{J} = \{J_1, \dots, J_n\}$ and $N_M$ machines $\mathcal{M} = \{M_1, \dots, M_m\}$.
            \item \textbf{Operations}: Each job $J_i$ consists of a sequence of operations $O_{i1} \to O_{i2} \dots \to O_{in_i}$.
            \item \textbf{Processing}: Operation $O_{ij}$ requires machine $\mu_{ij} \in \mathcal{M}$ for an uninterrupted duration $p_{ij}$.
            \item \textbf{Constraints}: 
            \begin{enumerate}
                \item \textit{Precedence}: $O_{i,j+1}$ cannot start until $O_{ij}$ completes.
                \item \textit{Capacity}: A machine processes at most one operation at a time.
            \end{enumerate}
            \item \textbf{Objective}: Minimize the \textbf{Makespan} $C_{max} = \max_{i,j} (S_{ij} + p_{ij})$, where $S_{ij}$ is the start time of $O_{ij}$.
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{Illustrative Example}
    Consider 2 Jobs and 2 Machines:
    \begin{itemize}
        \item \textbf{Job 1}: $O_{11}$ on $M_1$ ($p_{11}=2$), then $O_{12}$ on $M_2$ ($p_{12}=1$)
        \item \textbf{Job 2}: $O_{21}$ on $M_2$ ($p_{21}=1$), then $O_{22}$ on $M_1$ ($p_{22}=2$)
    \end{itemize}
    
    \pause

    \textbf{Optimal Schedule (Gantt Chart): Makespan = 3}

\section{Hardness of JSSP}

\begin{frame}{NP-Hardness Claim}
    \begin{block}{Theorem}
        The Job Shop Scheduling Problem is strongly \textbf{NP-Hard} for $N_M \ge 3$.
    \end{block}
    
    \pause
    \vspace{0.5em}
    \textbf{Proof via Polynomial Reduction from PARTITION}:
    \begin{itemize}[<+->]
        \item \textbf{PARTITION}: Given a multiset $S = \{a_1, \dots, a_n\}$ where $\sum a_i = 2B$. Can $S$ be partitioned into two subsets of exact sum $B$?
        \item \textbf{Reduction}: We construct a JSSP instance with $m=3$ machines and $n+1$ jobs, targeting a makespan of $C_{max} = 3B$.
        \item \textbf{The "Frame" Job (Job 0)}: Needs $M_1, M_2, M_3$ sequentially for time $B$ each. Total time = $3B$. It \textit{must} run continuously: $M_1 [0,B]$, $M_2 [B,2B]$, $M_3 [2B,3B]$.
        \item \textbf{The "Item" Jobs (Job $1 \dots n$)}: Job $i$ requires $M_2$ for time $a_i$.
    \end{itemize}
\end{frame}

\begin{frame}{NP-Hardness Illustration}
    Since $M_2$ is occupied by Job 0 from $[B, 2B]$, the Item Jobs (totaling $2B$) must fit perfectly into the two remaining slots: $[0, B]$ and $[2B, 3B]$.
    
    \vspace{1em}
    \centering
    \begin{tikzpicture}[x=1.2cm, y=0.8cm]
        \draw[thick,->] (0,0) -- (6.5,0) node[below] {Time};
        \draw[thick,->] (0,0) -- (0,3.5);
        \node at (-0.5, 0.5) {$M_3$}; \node at (-0.5, 1.5) {$M_2$}; \node at (-0.5, 2.5) {$M_1$};
        
        \foreach \x/\lbl in {0/0, 2/B, 4/2B, 6/3B} \draw (\x,0.1)--(\x,-0.1) node[below] {\lbl};
        
        % Frame Job
        \draw[fill=gray!40] (0, 2.1) rectangle (2, 2.9) node[pos=.5] {Job 0};
        \draw[fill=gray!40] (2, 1.1) rectangle (4, 1.9) node[pos=.5] {Job 0};
        \draw[fill=gray!40] (4, 0.1) rectangle (6, 0.9) node[pos=.5] {Job 0};
        
        % Gaps
        \draw[dashed, red, thick] (0, 1.1) rectangle (2, 1.9) node[pos=.5] {Subset 1};
        \draw[dashed, red, thick] (4, 1.1) rectangle (6, 1.9) node[pos=.5] {Subset 2};
    \end{tikzpicture}
    
    \vspace{1em}
    \pause
    \textbf{Conclusion}: Achieving $C_{max}=3B$ is possible \textit{if and only if} the items can be partitioned perfectly into two sets of size $B$. Thus, JSSP is NP-Hard.
\end{frame}

        \textbf{Metaheuristics} & \textbf{Simulated Annealing} & High quality (Kirkpatrick, 1983) \\ \hline

    \end{tabular}
    
    \pause
    \vspace{1em}
    \textbf{Chosen Algorithms for Implementation}:
    \begin{enumerate}
        \item \textbf{Branch \& Bound (Exact)}: To establish absolute baselines.
        \item \textbf{MWKR (Heuristic)}: A fast priority dispatching rule.
        \item \textbf{Simulated Annealing (Metaheuristic)}: For high-quality approximations.
    \end{enumerate}
\end{frame}

\section{Exact Algorithm: Branch \& Bound}

\begin{frame}{B\&B: Initialization \& Upper Bound}
    \scriptsize
    \begin{algorithmic}[1]
        \Function{Best\_Greedy}{$jobs, num\_jobs, num\_machines$}
            \State $best\_makespan \leftarrow \infty$
            \State $best\_schedule \leftarrow \text{NONE}$
            \For{$rule \in \{FIFO, SPT, LRPT, MOR\}$}
                \State $(ms, sched) \leftarrow \text{GREEDY}(jobs, \dots, rule)$
                \If{$ms < best\_makespan$}
                    \State $best\_makespan \leftarrow ms$
                    \State $best\_schedule \leftarrow sched$
                \EndIf
            \EndFor
            \State \Return $(best\_makespan, best\_schedule)$
        \EndFunction
        
        \Statex
        
        \Function{Solve\_Instance}{$jobs$}
            \State $(greedy\_ms, \_) \leftarrow \text{BEST\_GREEDY}(jobs, \dots)$
            \State \Return $\text{BRANCH\_AND\_BOUND}(jobs, \dots, greedy\_ms)$
        \EndFunction
    \end{algorithmic}
\end{frame}

\begin{frame}{B\&B: Lower Bound Function}
    \scriptsize
    \begin{algorithmic}[1]
        \Function{Lower\_Bound}{$jobs, job\_step, job\_avail, mach\_avail$}
            \State $lb\_job \leftarrow 0$
            \For{\text{each job } $j$}
                \State $job\_remaining \leftarrow$ Sum of $t$ for remaining operations of $j$
                \State $lb\_job \leftarrow \max(lb\_job, job\_avail[j] + job\_remaining)$
            \EndFor
            
            \Statex
            
            \State $lb\_machine \leftarrow 0$
            \For{\text{each machine } $m$}
                \State $m\_remaining \leftarrow$ Sum of $t$ for all operations requiring $m$
                \State $lb\_machine \leftarrow \max(lb\_machine, mach\_avail[m] + m\_remaining)$
            \EndFor
            
            \State \Return $\max(lb\_job, lb\_machine)$
        \EndFunction
    \end{algorithmic}
\end{frame}

\begin{frame}{B\&B: Main Loop \& Pruning}
    \scriptsize
    \begin{algorithmic}[1]
        \Function{Branch\_And\_Bound}{$jobs, \dots, upper\_bound$}
            \State $best \leftarrow upper\_bound$
            \State $Queue \leftarrow \text{empty min-heap}$
            \State \text{PUSH } $(\text{LOWER\_BOUND}(init\_state), init\_state) \text{ to } Queue$
            
            \While{$Queue \text{ is not empty}$}
                \State $(lb, state) \leftarrow \text{POP}(Queue)$
                
                \If{$lb \ge best$}
                    \State \textbf{continue} \Comment{Prune this branch}
                \EndIf
                
                \If{\text{all jobs finished in } $state$}
                    \State $makespan \leftarrow \max(state.job\_avail)$
                    \State $best \leftarrow \min(best, makespan)$
                    \State \textbf{continue}
                \EndIf
                
                \State \dots \Comment{Branching logic handled in next phase}
            \EndWhile
        \EndFunction
    \end{algorithmic}
\end{frame}

\begin{frame}{B\&B: State Generation (Branching)}
    \scriptsize 
    \begin{algorithmic}[1]
        \State \dots \Comment{Inside the While Loop}
        \For{\text{each job } $j$}
            \If{\text{job } $j \text{ is finished}$} \textbf{continue} \EndIf
            
            \State $(m, t) \leftarrow \text{next operation of job } j$
            \State $start \leftarrow \max(state.job\_avail[j], state.mach\_avail[m])$
            \State $finish \leftarrow start + t$
            
            \State $child \leftarrow \text{COPY}(state)$
            \State $child.job\_step[j] \mathrel{+}= 1$
            \State $child.job\_avail[j] \leftarrow finish$; \quad $child.mach\_avail[m] \leftarrow finish$
            
            \State $child\_lb \leftarrow \text{LOWER\_BOUND}(child)$
            
            \If{$child\_lb < best$}
                \State \text{PUSH } $(child\_lb, child) \text{ to } Queue$
            \EndIf
        \EndFor
    \end{algorithmic}
\end{frame}

\section{MWKR Algorithm}

\begin{frame}{MWKR Algorithm}
    \textbf{Most Work Remaining (MWKR)}: A greedy Priority Dispatching Rule (PDR) that schedules the operation belonging to the job with the maximum total remaining processing time.
    
    \vspace{0.5em}
    \scriptsize
    \begin{algorithmic}[1]
        \State $t \leftarrow 0$, $S \leftarrow$ \{First operations of all jobs\}
        \While{Not all ops scheduled}
            \State Find machine $M_k$ free at time $t$
            \State Let $A \subseteq S$ be operations requiring $M_k$
            \If{$A$ is empty}
                \State $t \leftarrow$ next completion time; \textbf{continue}
            \EndIf
            \State \textbf{Select} $O_{ij} \in A$ maximizing $\sum_{x=j}^{n_i} p_{ix}$
            \State Schedule $O_{ij}$; $S \leftarrow (S \setminus \{O_{ij}\}) \cup \{O_{i,j+1}\}$
        \EndWhile
    \end{algorithmic}
\end{frame}

\section{Simulated Annealing}

\begin{frame}{Simulated Annealing: Core Logic}
    Operates on the \textbf{Disjunctive Graph}. Edges are oriented to form a directed acyclic graph representing the schedule.
    
    \begin{itemize}[<+->]
        \item \textbf{State}: A valid topological sort of the operations.
        \item \textbf{Neighborhood}: Find the \textit{Critical Path} (the longest path defining the makespan). Swap two adjacent operations on this path that share the same machine.
        \item \textbf{Acceptance}: If $\Delta C_{max} < 0$, accept. If $\Delta C_{max} \ge 0$, accept with probability $P = \exp(-\Delta C_{max} / T)$.
    \end{itemize}
\end{frame}

\begin{frame}{Simulated Annealing Pseudocode}
    \scriptsize
    \begin{algorithmic}[1]
        \State $S \leftarrow \text{RandomSchedule()}$ 
        \State $T \leftarrow T_0$
        \While{$T > T_{final}$}
            \State $cp \leftarrow \text{FindCriticalPath}(S)$
            \State $S_{new} \leftarrow \text{SwapAdjacentMachineOps}(cp)$
            \State $\Delta = Makespan(S_{new}) - Makespan(S)$
            \If{$\Delta < 0$ \textbf{or} $Random(0,1) < e^{-\Delta/T}$}
                \State $S \leftarrow S_{new}$
            \EndIf
            \State $T \leftarrow \alpha \cdot T$
        \EndWhile
        \State \textbf{Return} Best schedule found
    \end{algorithmic}
\end{frame}

\section{Conclusions \& Limitations}

\begin{frame}{Conclusions}
    \begin{block}{Key Takeaways}
        \begin{itemize}[<+->]
            \item \textbf{MWKR} is blazing fast ($\sim 0.8$ ms) but highly sub-optimal ($\sim 70.6\%$ accuracy) due to myopic routing decisions.
            \item \textbf{Branch \& Bound} operates relatively quickly ($\sim 0.67$ s) by effectively pruning the search space, but early stopping constraints result in an average accuracy of $\sim 86.8\%$.
            \item \textbf{Simulated Annealing} achieves the highest overall accuracy ($\sim 88.2\%$) by successfully balancing exploration and exploitation, trading off with the highest computational cost ($\sim 2.06$ s).
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{Limitations \& Future Work}
    \textbf{Limitations \& Future Work}:
    \vspace{0.5em}
    \begin{itemize}[<+->]
        \item \textbf{Time-Bounded Branch \& Bound}: To keep experiments computationally feasible across 1,000 instances, we imposed a strict time limit on the B\&B algorithm. Increasing this timeout threshold would allow deeper exploration of the search tree, likely improving its accuracy well beyond $86.8\%$.
        \item \textbf{Empirical Parameter Tuning}: Hyperparameters for Simulated Annealing (cooling rate, initial temperature) were selected empirically and may not be globally optimal across all problem complexities.
        \item \textbf{Algorithmic Enhancements}: We evaluated standard implementations. Employing hybrid initialization (e.g., seeding SA with the MWKR schedule) or advanced pruning constraints for B\&B could further close the optimality gap.
    \end{itemize}
\end{frame}
